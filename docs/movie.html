Chapter 8 The Movie Dataset

Neo4J comes with an example dataset called movies, which you can use with:

play_movies() %>%
  call_neo4j(con)

res <- 'MATCH (per)-[act:ACTED_IN]->(mov) RETURN per, mov' %>% 
  call_neo4j(con)  
df <- bind_cols(
   purrr::pluck(res, "per"),
   purrr::pluck(res, "mov")
)
df  %>%
  count(name, sort = TRUE) %>%
  top_n(10)
